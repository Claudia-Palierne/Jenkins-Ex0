@Library('test_jenkins@main') _
pipeline {
    agent any
    parameters {
        string(name: 'BRANCH', defaultValue: 'master')
        string(name: 'REPOSITORY', defaultValue: 'selenium_jobs')
    }
    stages {
        stage('Clean workspace'){
            steps {
                cleanWs()
            }
        }
        stage('Print Param') {
            steps {
                echo params.BRANCH
                echo params.REPOSITORY
            }
        }
        stage('Git Checkout') {
            steps {
                gitCheckout(
                        branch: "${params.BRANCH}",
                        url: "https://github.com/Claudia-Palierne/${params.REPOSITORY}.git"
                )
            }
        }
        stage('Run'){
            steps{
                script{
                        bat "dir"
                        bat "pip install -r requirement.txt" 
                        bat "python DigitRecogition.py"
                }
            }
        }
        stage('Archive Images') {
            steps {
                archiveArtifacts artifacts: '**/*.png', allowEmptyArchive: true
            }
        }
        post {
            always {
                cleanWs()
            }
        }
    }
